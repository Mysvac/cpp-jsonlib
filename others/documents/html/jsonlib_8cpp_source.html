20 static void escapeUnicode(std::string& res, const std::string& str, size_t& index) {
21 // 将十六进制字符串转换为 Unicode 码点
22 if (index + 4 >= str.size()) throw JsonStructureException{ "Illegel string.\n" };
23 ++index;
24 std::istringstream iss(str.substr(index, 4));
25 index += 3;
26 unsigned int codePoint;
27 iss >> std::hex >> codePoint;
28
29 if (iss.fail() || codePoint > 0x10FFFF) {
30 std::cerr << "Illegel Unicode string.\n" << std::endl;
31 return;
32 }
33
34 if (codePoint >= 0xD800 && codePoint <= 0xDFFF) {
35 if (codePoint >= 0xDC00) {
36 std::cerr << "Illegel Unicode string.\n" << std::endl;
37 return;
38 }
39
40 // 检查下一个转义序列是否是低代理
41 if (index + 6 >= str.length() || str[index + 1] != '\\' || str[index + 2] != 'u') {
42 std::cerr << "Illegel Unicode string.\n" << std::endl;
43 return;
44 }
45
46 // 解析低代理
47 index += 3;
48 std::string lowHexStr = str.substr(index, 4);
49 index += 3; // 将索引移动到低代理转义序列之后
50
51 unsigned int lowCodePoint;
52 std::istringstream lowIss(lowHexStr);
53 lowIss >> std::hex >> lowCodePoint;
54
55 if (lowIss.fail() || lowCodePoint < 0xDC00 || lowCodePoint > 0xDFFF) {
56 std::cerr << "Illegel Unicode string.\n" << std::endl;
57 return;
58 }
59
60 // 将代理对组合为单个码点
61 codePoint = 0x10000 + ((codePoint - 0xD800) << 10) + (lowCodePoint - 0xDC00);
62 }
63
64 // 将码点编码为 UTF-8
65 if (codePoint <= 0x7F) {
66 res += static_cast<char>(codePoint);
67 }
68 else if (codePoint <= 0x7FF) {
69 res += static_cast<char>(0xC0 | (codePoint >> 6));
70 res += static_cast<char>(0x80 | (codePoint & 0x3F));
71 }
72 else if (codePoint <= 0xFFFF) {
73 res += static_cast<char>(0xE0 | (codePoint >> 12));
74 res += static_cast<char>(0x80 | ((codePoint >> 6) & 0x3F));
75 res += static_cast<char>(0x80 | (codePoint & 0x3F));
76 }
77 else if (codePoint <= 0x10FFFF) {
78 res += static_cast<char>(0xF0 | (codePoint >> 18));
79 res += static_cast<char>(0x80 | ((codePoint >> 12) & 0x3F));
80 res += static_cast<char>(0x80 | ((codePoint >> 6) & 0x3F));
81 res += static_cast<char>(0x80 | (codePoint & 0x3F));
82 }
83 else {
84 // return;
85 std::cerr << "Illegel Unicode string.\n" << std::endl;
86 return;
87 }

94 static bool is_number(const std::string& str) noexcept {
95 size_t it = 0;
96 size_t len = str.size();
97 if (str[0] == '-') {
98 if (len == 1 && !std::isdigit(str[1])) return false;
99 ++it;
100 }
101 bool have_not_point = true;
102 bool have_not_e = true;
103 while (it < len) {
104 if (std::isdigit(str[it])) ++it;
105 else if (str[it] == '.' && have_not_point && have_not_e) {
106 have_not_point = false;
107 ++it;
108 }
109 else if ((str[it] == 'e' || str[it] == 'E') && have_not_e) {
110 have_not_e = false;
111 ++it;
112 if (it >= len) return false;
113 if ((it == '-' || it == '+') && it < len - 1) ++it;
114 }
115 else return false;
116 }
117
118#ifdef JSONLIB_CHECK_START_ZERO
119 if (str[0] == '0' && have_not_point) throw JsonStructureException{"Intager could not start with 0.\n"};
120#endif
121
122 return true;
123 }

129 std::string res;
130 // 提前分配空间，减少扩容开销
131 if (str.size() > 13) res.reserve(str.size() + 3 + (str.size() >> 4));
132 res += "\"";

143 case '\f':

147 res += "\\t";

150 res += "\\r";
151 break;

158 }

167 static std::string escape(const std::string& str, const size_t& left, const size_t& right) {
168 std::string res;
169 // 预分配空间
170 if (right - left > 14) res.reserve(right - left + 1);
171
172 size_t index = left;
173 while (index <= right) {
174 if (str[index] == '\\') {
175 // 转义字符处理
176 ++index;
177 if (index > right) throw JsonStructureException{ "String end with escape character :" + str.substr(left, right - left + 1) };
178 switch (str[index])
179 {
180 case '\"':
181 res += '\"';
182 break;
183 case '\\':
184 res += '\\';
185 break;
186 case '/':
187 res += '/';
188 break;
189 case 'n':
190 res += '\n';
191 break;
192 case 'r':
193 res += '\r';
194 break;
195 case 't':
196 res += '\t';
197 break;
198 case 'f':
199 res += '\f';
200 break;
201 case 'b':
202 res += '\b';
203 break;
204 case 'u':
205 escapeUnicode(res, str, index);
206 // res += "\\u";
207 break;
208 default:
209 throw JsonStructureException{ "Illegal escape characters " };
210 break;
211 }
212 }
213 else res += str[index];
214 ++index;
215 }
216 return std::move(res);

309 // 辅助递归构造函数
310 JsonBasic::JsonBasic(const std::string& str, size_t& index, const size_t& tail) {
311 switch (str[index]) {
312 case '{':
313 {
314 type_ = JsonType::OBJECT;
315 content_ = Map{};
316 Map& map = std::get<Map>(content_);
317 std::string key;
318
319#ifdef JSONLIB_CHECK_END_COMMA
320 unsigned int num_comma = 0;
321#endif
322 ++index;
323 while (str[index] != '}') {
324 while (std::isspace(str[index])) ++index;
325 // 空集合，结束
326 if (str[index] == '}') break;
327 if (str[index] != '\"') throw JsonStructureException{ "Key is not string.\n" };
328 // 获取键
329 key = Json::escape_next(str, index, tail);
330 ++index;
331 // 分隔符 子元素 不可能以 '\"' 结尾，无需检测越界
332 while (std::isspace(str[index])) ++index;
333 if (str[index] == ':') ++index;
334 else throw JsonStructureException{ "Unknow element.\n" };
335
336 // 获取值 子元素 不可能以':'结尾，此处不检测
337 while (std::isspace(str[index])) ++index;
338 map[std::move(key)] = JsonBasic{ str, index, tail };
339
340 // index默认停在子元素的最后一个字符，所以需要先自增
341 ++index;
342 while (index < tail && std::isspace(str[index])) ++index;
343 // 判断合法性，结束后下个字符要么逗号，要么终止
344 if (str[index] == ',')
345#ifdef JSONLIB_CHECK_END_COMMA
346 ++index, ++num_comma;
347#else
348 ++index;
349#endif
350 else if (str[index] != '}') throw JsonStructureException{ "Unknow element.\n" };
351 }
352
353
354#ifdef JSONLIB_CHECK_END_COMMA
355 if (num_comma == map.size() && num_comma) throw JsonStructureException{ "Redundant comma.\n" };
356#endif
357
358 // 大于等于，说明内部括号未闭合
359 if (index >= tail) throw JsonStructureException{ "The { } are not closed.\n" };
360 }
361 break;
362 case '[':
363 {
364 type_ = JsonType::ARRAY;
365 content_ = List{};
366 List& list = std::get<List>(content_);
367
368#ifdef JSONLIB_CHECK_END_COMMA
369 unsigned int num_comma = 0;
370#endif
371
372 ++index;
373 while (str[index] != ']') {
374 while (std::isspace(str[index])) ++index;
375 // 空列表，结束
376 if (str[index] == ']') break;
377 // 添加子类并移动index到子类末尾
378 list.push_back(JsonBasic{ str, index, tail });
379 // index默认停在子元素的最后一个字符，所以需要先自增
380 ++index;
381 while (index < tail && std::isspace(str[index])) ++index;
382 // 判断合法性，结束后下个字符要么逗号，要么终止
383 if (str[index] == ',')
384#ifdef JSONLIB_CHECK_END_COMMA
385 ++index, ++num_comma;
386#else
387 ++index;
388#endif
389 else if (str[index] != ']') throw JsonStructureException{ "Unknow element.\n" };
390 }
391#ifdef JSONLIB_CHECK_END_COMMA
392 if (num_comma == list.size() && num_comma) throw JsonStructureException{ "Redundant comma.\n" };
393#endif
394 // 大于等于，说明内部括号未闭合
395 if (index >= tail) throw JsonStructureException{ "The [ ] are not closed.\n" };
396 }
397 break;
398 case '\"':
399 type_ = JsonType::STRING;
400 content_ = Json::escape_next(str, index, tail);
401 break;
402 case 't':
403 // str.compare return 0 if equal
404 if (tail - index <= 3 || str.compare(index, 4, "true")) throw JsonStructureException{ "Unknow element.\n" };
405 type_ = JsonType::BOOLEN;
406 content_ = std::string{ "true" };
407 index += 3;
408 break;
409 case 'f':
410 if (tail - index <= 4 || str.compare(index, 5, "false")) throw JsonStructureException{ "Unknow element.\n" };
411 type_ = JsonType::BOOLEN;
412 content_ = std::string{ "false" };
413 index += 4;
414 break;
415 case 'n':
416 if (tail - index <= 4 || str.compare(index, 4, "null")) throw JsonStructureException{ "Unknow element.\n" };
417 // type_ = JsonType::ISNULL;
418 index += 3;
419 break;
420 default:
421 if (!std::isdigit(str[index]) && str[index] != '-') throw JsonStructureException{ "Unknow element.\n" };
422 if (!std::isdigit(str[index + 1]) && str[index] == '-') throw JsonStructureException{ "Unknow element.\n" };
423 {
424 type_ = JsonType::NUMBER;
425 bool have_not_point = true;
426 std::string tmp;
427 tmp += str[index];
428
429#ifdef JSONLIB_CHECK_START_ZERO
430 const char head_char= str[index];
431#endif
432 while (index < tail)
433 {
434 ++index;
435 while (std::isdigit(str[index])) {
436 tmp += str[index];
437 ++index;
438 }
439 if (str[index] == '.') {
440 tmp += str[index];
441 ++index;
442 while (std::isdigit(str[index])) {
443 tmp += str[index];
444 ++index;
445 }
446 }
447 if (str[index] == 'e' || str[index] == 'E') {
448 tmp += str[index];
449 ++index;
450 if (str[index] == '-' || str[index] == '+') {
451 tmp += str[index];
452 ++index;
453 }
454 while (std::isdigit(str[index])) {
455 tmp += str[index];
456 ++index;
457 }
458 }
459 --index;
460 break;
461 }
462
463#ifdef JSONLIB_CHECK_START_ZERO
464 if (head_char == '0' && have_not_point) throw JsonStructureException{ "Intager could not start with 0.\n" };
465#endif
466 content_ = std::move(tmp);
467 }
468 break;
469 }

472 // 字符串反序列化构造函数
473 JsonBasic::JsonBasic(const std::string& str) {
474 // 起始位置
475 size_t index = 0;
476 // 字符串长度
477 const size_t len = str.size();
478 // 末尾位置
479 size_t tail = len - 1;
480
481 while (index < len && std::isspace(str[index])) ++index;
482 if (index == len) throw JsonStructureException{ "Empty JSON text.\n" };
483 while (std::isspace(str[tail])) --tail;
484
485 switch (str[index]) {
486 case '{':
487 if (str[tail] != '}') throw JsonStructureException{ "The { } are not closed.\n" };
488 {
489 type_ = JsonType::OBJECT;
490 content_ = Map{};
491 Map& map = std::get<Map>(content_);
492 std::string key;
493
494#ifdef JSONLIB_CHECK_END_COMMA
495 unsigned int num_comma = 0;
496#endif
497 ++index;
498 while (index < tail) {
499 while (std::isspace(str[index])) ++index;
500 // 空集合，结束
501 if (index == tail) break;
502 if (str[index] != '\"') throw JsonStructureException{ "Unknow element.\n" };
503 // 获取键
504 key = Json::escape_next(str, index, tail);
505 ++index;
506 // 分隔符
507 while (std::isspace(str[index])) ++index;
508 if (str[index] == ':') ++index;
509 else throw JsonStructureException{ "Unknow element.\n" };
510
511 // 获取值
512 while (std::isspace(str[index])) ++index;
513 map[std::move(key)] = JsonBasic{ str, index, tail };
514
515 // index默认停在子元素的最后一个字符，所以需要先自增
516 ++index;
517 while (std::isspace(str[index])) ++index;
518 // 判断合法性，结束后下个字符要么逗号，要么终止
519 if (str[index] == ',')
520#ifdef JSONLIB_CHECK_END_COMMA
521 ++index, ++num_comma;
522#else
523 ++index;
524#endif
525 else if (index != tail) throw JsonStructureException{ "Unknow element.\n" };
526 }
527
528#ifdef JSONLIB_CHECK_END_COMMA
529 if (num_comma == map.size() && num_comma) throw JsonStructureException{ "Redundant comma.\n" };
530#endif
531 // 大于，说明大括号未闭合
532 if (index > tail) throw JsonStructureException{ "The { } are not closed.\n" };
533 }
534 break;
535 case '[':
536 if (str[tail] != ']') throw JsonStructureException{ "The [ ] are not closed.\n" };
537 {
538 type_ = JsonType::ARRAY;
539 content_ = List{};
540 List& list = std::get<List>(content_);
541
542#ifdef JSONLIB_CHECK_END_COMMA
543 unsigned int num_comma = 0;
544#endif
545
546 ++index;
547 while (index < tail) {
548 while (std::isspace(str[index])) ++index;
549 // 空列表，结束
550 if (index == tail) break;
551 // 添加子类并移动index到子类末尾
552 list.push_back(JsonBasic{ str, index, tail });
553 // index默认停在子元素的最后一个字符，所以需要先自增
554 ++index;
555 while (std::isspace(str[index])) ++index;
556 // 判断合法性，结束后下个字符要么逗号，要么终止
557 if (str[index] == ',')
558#ifdef JSONLIB_CHECK_END_COMMA
559 ++index, ++num_comma;
560#else
561 ++index;
562#endif
563 else if (index != tail) throw JsonStructureException{ "Unknow element.\n" };
564 }
565#ifdef JSONLIB_CHECK_END_COMMA
566 if (num_comma == list.size() && num_comma) throw JsonStructureException{ "Redundant comma.\n" };
567#endif
568 // 大于，说明方括号未闭合
569 if (index > tail) throw JsonStructureException{ "The [ ] are not closed.\n" };
570 }
571 break;
572 case '\"':
573 if (str[tail] != '\"') throw JsonStructureException{ "The \" \" are not closed.\n" };
574 type_ = JsonType::STRING;
575 content_ = Json::escape(str, index + 1, tail - 1);
576 break;
577 case 't':
578 // str.compare return 0 if equal
579 if (tail - index <= 3 || str.compare(index, 4, "true")) throw JsonStructureException{ "Unknow element.\n" };
580 type_ = JsonType::BOOLEN;
581 content_ = std::string{ "true" };
582 break;
583 case 'f':
584 if (tail - index <= 4 || str.compare(index, 5, "false")) throw JsonStructureException{ "Unknow element.\n" };
585 type_ = JsonType::BOOLEN;
586 content_ = std::string{ "false" };
587 break;
588 case 'n':
589 if (tail - index <= 4 || str.compare(index, 4, "null")) throw JsonStructureException{ "Unknow element.\n" };
590 // type_ = JsonType::ISNULL;
591 break;
592 default:
593 if (!std::isdigit(str[index]) && str[index] != '-' && str[index] != '+') throw JsonStructureException{ "Unknow element.\n" };
594 if (!std::isdigit(str[index + 1]) && (str[index] == '-' || str[index] == '+')) throw JsonStructureException{ "Unknow element.\n" };
595 {
596 std::string tmp = str.substr(index, tail - index + 1);
597 if (!Json::is_number(tmp)) throw JsonStructureException{ "Unknow element.\n" };
598 type_ = JsonType::NUMBER;
599 content_ = std::move(tmp);
600 }
601 break;
602 }

679 // 序列化JSON对象
680 std::string JsonBasic::serialize() const noexcept {
681 switch (type_)
682 {
683 case JsonType::OBJECT:
684 {
685 // 对象类型
686 // 是否在开头加上逗号
687 std::string res{ "{" };
688 const Map& map = std::get<Map>(content_);
689 for (const std::pair<std::string, JsonBasic>& it : map) {
690 // 键是字符串，需要反转义
691 res += reverse_escape(it.first);
692 res += ':';
693 // 递归序列号
694 res += it.second.serialize();
695 res += ',';
696 }
697 if (*res.rbegin() == ',') *res.rbegin() = '}';
698 else res += '}';
699 return std::move(res);
700 }
701 break;
702 case JsonType::ARRAY:
703 {
704 // 数组类型
705 // 是否在开头加上逗号
706 std::string res{ "[" };
707 const List& list = std::get<List>(content_);
708 for (const JsonBasic& it : list) {
709 // 递归序列号
710 res += it.serialize();
711 res += ',';
712 }
713 if (*res.rbegin() == ',') *res.rbegin() = ']';
714 else res += ']';
715 return std::move(res);
716 }

729 // 序列化JSON对象含空格和换行
730 std::string JsonBasic::serialize_pretty(const size_t& space_num, const size_t& depth) const noexcept {
731 switch (type_)
732 {
733 case JsonType::OBJECT:
734 {
735 std::string tabs(depth * space_num, ' ');
736 // 对象类型
737 std::string res{ "{" };
738
739 const Map& map = std::get<Map>(content_);
740 for (const std::pair<std::string, JsonBasic>& it : map) {
741 res += '\n' + std::string(space_num, ' ') + tabs;
742 // 键是字符串，需要反转义
743 res += reverse_escape(it.first);
744 res += ": ";
745 // 递归序列号
746 res += it.second.serialize_pretty(space_num, depth + 1);
747 res += ',';
748 }
749 if (*res.rbegin() == ',') *res.rbegin() = '\n';
750 res += tabs + '}';
751 return std::move(res);
752 }
753 break;
754 case JsonType::ARRAY:
755 {
756 std::string tabs(depth * space_num, ' ');
757 // 数组类型
758 std::string res{ "[" };
759
760 const List& list = std::get<List>(content_);
761 for (const JsonBasic& it : list) {
762 // 递归序列号
763 res += '\n' + std::string(space_num, ' ') + tabs;
764 res += it.serialize_pretty(space_num, depth + 1);
765 res += ',';
766 }
767 if (*res.rbegin() == ',') *res.rbegin() = '\n';
768 res += tabs + ']';
769 return std::move(res);
770 }
771 break;
772 case JsonType::STRING:
773 // 字符串类型，直接反转义，然后输出
774 return reverse_escape(std::get<std::string>(content_));
775 break;
776 default:
777 // 非对象 非列表 非字符串，可以直接返回内容
778 return std::get<std::string>(content_);
779 break;
780 }
